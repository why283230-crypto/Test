local function pressKey(keyCode)
    pcall(function()
        VirtualInputManager:SendKeyEvent(true,  keyCode, false, game)
        task.wait(0.02)
        VirtualInputManager:SendKeyEvent(false, keyCode, false, game)
    end)

local function findProximityPrompt(model)
    if not model then return nil end
    for _, v in ipairs(model:GetDescendants()) do
        if v:IsA("ProximityPrompt") then return v end
    end
    return nil
end

local function triggerPrompt(prompt)
    if not prompt then return false end
    pcall(function()
        if fireproximityprompt then
            fireproximityprompt(prompt)
        else
            pressKey(prompt.KeyboardKeyCode ~= Enum.KeyCode.Unknown and prompt.KeyboardKeyCode or Enum.KeyCode.E)
        end
    end)
    return true
end

local function getBossPosition(bossName)
    local folderName = bossPaths[bossName]
    if not folderName then return nil end
    local clientEntities = workspace:FindFirstChild("ClientEntities")
    if not clientEntities then return nil end
    local bossModel = clientEntities:FindFirstChild(folderName)
    if not bossModel then return nil end
    local root = bossModel:FindFirstChild("HumanoidRootPart") or bossModel:FindFirstChild("RootPart") or bossModel.PrimaryPart
    if not root then
        for _, part in ipairs(bossModel:GetDescendants()) do
            if part:IsA("BasePart") then root = part break end
        end
    end
    return root and root.Position
end

local function checkAnyBossActive()
    for bossName, enabled in pairs(targetBosses) do
        if enabled and getBossPosition(bossName) then
            return true
        end
    end
    return false
end

-- ========================
-- Slot Stroke Engine
-- ========================
local function getSlotStroke()
    local ok, stroke = pcall(function()
        local playerGui = LocalPlayer:FindFirstChild("PlayerGui")
        local hotbar    = playerGui and playerGui:FindFirstChild("Main") and playerGui.Main:FindFirstChild("HUD") and playerGui.Main.HUD:FindFirstChild("Hotbar")
        if not hotbar then return nil end
        local slot = nil
        local slotName = hotbarSlotNames[selectedSlot]
        if slotName then slot = hotbar:FindFirstChild(slotName) end
        if not slot then
            for _, child in ipairs(hotbar:GetChildren()) do
                if child:IsA("GuiObject") and child.LayoutOrder == selectedSlot then
                    slot = child break
                end
            end
        end
        if not slot then
            local validSlots = {}
            for _, child in ipairs(hotbar:GetChildren()) do
                if child:IsA("ImageButton") or child:IsA("TextButton") or child.Name:lower():find("slot") then
                    table.insert(validSlots, child)
                end
            end
            table.sort(validSlots, function(a, b) return (a.LayoutOrder or 0) < (b.LayoutOrder or 0) end)
            slot = validSlots[selectedSlot]
        end
        return slot and slot:FindFirstChildOfClass("UIStroke") or nil
    end)
    return ok and stroke or nil
end

local function isStrokeBlack(stroke)
    if not stroke then return false end
    local c = stroke.Color
    return c.R < 0.1 and c.G < 0.1 and c.B < 0.1
end

local function tryGrabFruit(fruitName)
    local model = workspace:FindFirstChild(fruitName)
    if not model then return false end
    local hrp = getHRP()
    if not hrp then return false end

    local targetPart = model.PrimaryPart or model:FindFirstChild("HumanoidRootPart") or model:FindFirstChild("RootPart")
    if not targetPart then
        for _, part in ipairs(model:GetDescendants()) do
            if part:IsA("BasePart") then targetPart = part break end
        end
    end
    if not targetPart then return false end

    hrp.CFrame = CFrame.new(targetPart.Position + Vector3.new(0, 3, 0))
    WindUI:Notify({ Title = "Found: " .. fruitName, Content = "Teleported! Waiting 3s before grabbing...", Duration = 3, Icon = "zap" })
    task.wait(3)

    local stillExists = workspace:FindFirstChild(fruitName)
    if not stillExists then
        WindUI:Notify({ Title = fruitName .. " disappeared!", Duration = 2, Icon = "x" })
        return false
    end

    local prompt = findProximityPrompt(stillExists)
    if prompt then
        triggerPrompt(prompt)
        WindUI:Notify({ Title = "Grabbed: " .. fruitName, Content = "ProximityPrompt triggered!", Duration = 3, Icon = "check-circle" })
        return true
    else
        WindUI:Notify({ Title = "No Prompt on " .. fruitName, Duration = 2, Icon = "alert-circle" })
        return false
    end
end

local function getAnyPendingFruitModel()
    for _, fruit in ipairs(fruitTargets) do
        if fruit.enabled then
            local model = workspace:FindFirstChild(fruit.name)
            if model then return fruit.name end
        end
    end
    return nil
end

local function grabAllPendingFruits()
    for _, fruit in ipairs(fruitTargets) do
        if fruit.enabled then
            local model = workspace:FindFirstChild(fruit.name)
            if model then
                tryGrabFruit(fruit.name)
                task.wait(1)
            end
        end
    end
end

-- Main Boss Farm Engine (Priority 1)
-- ========================
local function updateFarmState()
    local anyBossEnabled = false
    for _, enabled in pairs(targetBosses) do
        if enabled then anyBossEnabled = true break end
    end

    if not anyBossEnabled then
        farmingEnabled = false
        isBossActiveNow = false
        if farmThread then task.cancel(farmThread) farmThread = nil end
        return
    end

    if farmingEnabled then return end
    farmingEnabled = true

    farmThread = task.spawn(function()
        while farmingEnabled do
            if trainX4Enabled then
                task.wait(0.5)
                continue
            end

            local hasTarget = false
            local activeBossCount = 0

            for bossName, enabled in pairs(targetBosses) do
                if not enabled then continue end
                activeBossCount = activeBossCount + 1
                local hrp = getHRP()
                local bossPos = getBossPosition(bossName)
                if hrp and bossPos then
                    isBossActiveNow = true
                    hrp.CFrame = CFrame.new(bossPos)
                    hasTarget = true
                    hasResetThisCycle = false
                    task.wait(0.2)
                end
            end

            if not hasTarget then
                isBossActiveNow = false
                if activeBossCount > 0 then
                    if autoResetEnabled and not hasResetThisCycle then
                        local char = LocalPlayer.Character
                        local hum = char and char:FindFirstChildOfClass("Humanoid")
                        if hum and hum.Health > 0 then
                            hasResetThisCycle = true
                            hum.Health = 0
                            task.wait(0.5)
                        end
                    end
                    if useSavePosEnabled and savedPosition then
                        local hrp = getHRP()
                        if hrp then hrp.CFrame = CFrame.new(savedPosition) task.wait(0.3) end
                    end
                    task.wait(0.3)
                else
                    task.wait(0.1)
                end
            else
                task.wait(0.1)
            end
        end
    end)
end

local function scanClientEntities()
    local mobsFound = {}
    local clientEntities = workspace:FindFirstChild("ClientEntities")
    if clientEntities then
        for _, child in ipairs(clientEntities:GetChildren()) do
            if child:IsA("Model") and not table.find(mobsFound, child.Name) and not bossPaths[child.Name] and child.Name ~= "Kurama" then
                table.insert(mobsFound, child.Name)
            end
        end
    end
    if #mobsFound == 0 then table.insert(mobsFound, "No Mobs Found") end
    table.sort(mobsFound)
    availableMobs = mobsFound
    return mobsFound
end

local function getMobPosition(mobName)
    local clientEntities = workspace:FindFirstChild("ClientEntities")
    if not clientEntities then return nil end
    for _, child in ipairs(clientEntities:GetChildren()) do
        if child.Name == mobName and child:IsA("Model") then
            local root = child:FindFirstChild("HumanoidRootPart") or child:FindFirstChild("RootPart") or child.PrimaryPart
            if not root then
                for _, part in ipairs(child:GetDescendants()) do
                    if part:IsA("BasePart") then root = part break end
                end
            end
            if root then return root.Position end
        end
    end
    return nil
end

local function updateMobFarmState()
    if not farmMobEnabled then
        if farmMobThread then task.cancel(farmMobThread) farmMobThread = nil end
        return
    end
    farmMobThread = task.spawn(function()
        while farmMobEnabled do
            -- เช็คระบบความสำคัญ: 1. Auto Boss, 2. Auto Kurama, 3. Auto Fruit ต้องไม่ทำงานอยู่/ไม่มีเป้าหมาย
            if checkAnyBossActive() or isKuramaInArena or getAnyPendingFruitModel() or trainX4Enabled then
                task.wait(0.5)
                continue
            end
            
            local teleportsDone = false
            for mobName, isSelected in pairs(selectedMobs) do
                if not farmMobEnabled then break end
                if isSelected and mobName ~= "No Mobs Found" then
                    local pos = getMobPosition(mobName)
                    local hrp = getHRP()
                    if hrp and pos then
                        hrp.CFrame = CFrame.new(pos + Vector3.new(0, 5, 0))
                        teleportsDone = true
                        task.wait(0.4)
                    end
                end
            end
            if not teleportsDone then task.wait(0.5) else task.wait(0.1) end
        end
    end)
end

local function stopCameraLock()
    if camLockConn then camLockConn:Disconnect() camLockConn = nil end
end

local function startCameraLock()
    stopCameraLock()
    camLockConn = RunService.RenderStepped:Connect(function()
        if not isKuramaInArena then return end
        local kuramaCFrame = getKuramaPosition()
        local hrp = getHRP()
        if kuramaCFrame and hrp then
            local lookCFrame = CFrame.lookAt(hrp.Position, kuramaCFrame.Position)
            hrp.CFrame = CFrame.new(hrp.Position) * (lookCFrame - lookCFrame.Position)
            Camera.CFrame = CFrame.lookAt(Camera.CFrame.Position, kuramaCFrame.Position)
        end
    end)
end

local function stopAutoKurama()
    autoKuramaEnabled = false
    isKuramaInArena   = false
    stopCameraLock()
    stopMouseClick()
    stopKuramaSkill()
    if autoKuramaThread then task.cancel(autoKuramaThread) autoKuramaThread = nil end
end

local function startAutoKurama()
    stopAutoKurama()
    autoKuramaEnabled = true
    if GetValue("kuramaSkill", false) then startKuramaSkill() end

    autoKuramaThread = task.spawn(function()
        while autoKuramaEnabled do
            -- เช็คระบบความสำคัญ: 1. Auto Boss หากทำงานอยู่ ให้หยุดรอคุรากะแป๊บนึง
            if checkAnyBossActive() then
                task.wait(0.5)
                continue
            end

            isKuramaInArena = false
            local hrp = getHRP()
            local totemOk, totemObj = pcall(function() return workspace.Map.Arenas.Boss.KuramaArena.Totem end)

            if hrp and totemOk and totemObj then
                local totemPart = nil
                if totemObj:IsA("BasePart") then totemPart = totemObj else totemPart = totemObj:FindFirstChildWhichIsA("BasePart", true) end

                if totemPart then
                    hrp.CFrame = CFrame.new(totemPart.Position + Vector3.new(0, 3, 0))
                    WindUI:Notify({ Title = "Kurama Farm", Content = "Teleported to Totem, triggering...", Duration = 2, Icon = "zap" })
                    task.wait(1.5)
                end

                local prompt = findProximityPrompt(totemObj:IsA("BasePart") and totemObj.Parent or totemObj)
                if not prompt then pressKey(Enum.KeyCode.E) else triggerPrompt(prompt) end

                local waited = 0
                local foundKurama = false
                while waited < 20 and autoKuramaEnabled do
                    if checkAnyBossActive() then break end -- หลุดลูปถ้าบอสเกิดและเปิดออโต้บอสไว้
                    if getKuramaPosition() then foundKurama = true break end
                    task.wait(0.5)
                    waited = waited + 0.5
                end

                if checkAnyBossActive() then continue end

                if not foundKurama then
                    WindUI:Notify({ Title = "Kurama", Content = "Kurama model not found after 20s, retrying...", Duration = 3, Icon = "alert-circle" })
                    task.wait(1)
                    continue
                end
            else
                WindUI:Notify({ Title = "Kurama", Content = "Totem not found, retrying...", Duration = 2, Icon = "alert-circle" })
                task.wait(3)
                continue
            end
            
            if not autoKuramaEnabled or checkAnyBossActive() then break end

            isKuramaInArena = true
            startCameraLock()
            startMouseClick()
            WindUI:Notify({ Title = "Kurama Farm", Content = "Kurama found! Farming now...", Duration = 3, Icon = "check-circle" })

            local missCount = 0
            while autoKuramaEnabled do
                if checkAnyBossActive() then -- บอสเกิดกะทันหัน ให้ยกเลิกการตีคุรากะชั่วคราวเพื่อไปตีบอสก่อน
                    isKuramaInArena = false
                    stopCameraLock()
                    stopMouseClick()
                    break 
                end

                local kuramaCFrame = getKuramaPosition()
                local fightHRP     = getHRP()

                if kuramaCFrame and fightHRP then
                    missCount = 0
                    local behindPos = kuramaCFrame * CFrame.new(0, 40, 80)
                    fightHRP.CFrame = behindPos
                    task.wait(0.2)
                else
                    missCount = missCount + 1
                    if missCount >= 5 then
                        WindUI:Notify({ Title = "Kurama", Content = "Kurama model lost, restarting...", Duration = 3, Icon = "refresh-cw" })
                        break
                    end
                    task.wait(0.3)
                end
            end

            isKuramaInArena = false
            stopCameraLock()
            stopMouseClick()
            if autoKuramaEnabled then task.wait(2) end
        end
    end)
end

local function serverHop()
    WindUI:Notify({ Title = "Server Hop", Content = "Finding a new server...", Duration = 3, Icon = "refresh-cw" })
    task.wait(1)
    pcall(function()
        local servers = HttpService:JSONDecode(game:HttpGet(
            "https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=Asc&limit=100"
        ))
        for _, server in ipairs(servers.data) do
            if server.playing < server.maxPlayers and server.id ~= game.JobId then
                TeleportService:TeleportToPlaceInstance(game.PlaceId, server.id, LocalPlayer)
                break
            end
        end
    end)
end

local function stopAutoHop()

local function startAutoHop()

    autoHopThread = task.spawn(function()
        while autoHopEnabled do
            local waitTime = (hopInterval and hopInterval > 0) and (hopInterval * 60) or 600
            task.wait(waitTime)
            if not autoHopEnabled then break end
            serverHop()
        end
    end)
end

-- Teleport Zone Helpers

local function scanTrainingZones()
    local zones = {}
    local ok, trainingFolder = pcall(function() return workspace.Zones.Training end)
    if ok and trainingFolder then
        for _, child in ipairs(trainingFolder:GetChildren()) do table.insert(zones, child.Name) end
    end
    if #zones == 0 then table.insert(zones, "No Zones Found") end
    table.sort(zones)
    availableZones = zones
    return zones
end

local function getZonePosition(zoneName)
    local ok, trainingFolder = pcall(function() return workspace.Zones.Training end)
    if not ok or not trainingFolder then return nil end
    local zoneObj = trainingFolder:FindFirstChild(zoneName)
    if not zoneObj then return nil end
    if zoneObj:IsA("BasePart") then return zoneObj.Position end
    local root = zoneObj:FindFirstChild("HumanoidRootPart") or zoneObj:FindFirstChild("RootPart") or (zoneObj.PrimaryPart)
    if not root then
        for _, part in ipairs(zoneObj:GetDescendants()) do if part:IsA("BasePart") then root = part break end end
    end
    return root and root.Position or nil
end

-- ========================
-- Train x4 Clan Point Logic
-- ========================
local function stopTrainX4()
    trainX4Enabled = false
    lockedTrainPos = nil
    if trainX4Thread then task.cancel(trainX4Thread) trainX4Thread = nil end
end

local function startTrainX4()
    stopTrainX4()
    local hrp = getHRP()
    if not hrp then return end
    lockedTrainPos = hrp.CFrame
    trainX4Enabled = true
    pressKey(Enum.KeyCode.Four)
    trainX4Thread = task.spawn(function()
        local char = LocalPlayer.Character
        local hum  = char and char:FindFirstChildOfClass("Humanoid")
        while trainX4Enabled do
            local curHRP = getHRP()
            if curHRP and lockedTrainPos then curHRP.CFrame = lockedTrainPos end
            pressKey(Enum.KeyCode.W)
            task.wait(0.1)
            if hum then hum.Jump = true else pressKey(Enum.KeyCode.Space) end
            task.wait(0.4)
        end
    end)
end
