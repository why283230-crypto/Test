local function getHRP()
    local char = LocalPlayer.Character

local function pressKey(keyCode)
    pcall(function()
        VirtualInputManager:SendKeyEvent(true,  keyCode, false, game)
        task.wait(0.02)
        VirtualInputManager:SendKeyEvent(false, keyCode, false, game)
    end)
end

local function getBossPosition(bossName)
    local folderName = bossPaths[bossName]
    if not folderName then return nil end
    local clientEntities = workspace:FindFirstChild("ClientEntities")
    if not clientEntities then return nil end
    local bossModel = clientEntities:FindFirstChild(folderName)
    if not bossModel then return nil end
    local root = bossModel:FindFirstChild("HumanoidRootPart")
        or bossModel:FindFirstChild("RootPart")
        or bossModel.PrimaryPart
    if not root then
        for _, part in ipairs(bossModel:GetDescendants()) do
            if part:IsA("BasePart") then root = part break end
        end
    end
    return root and root.Position
end

-- ========================
-- getSlotStroke
-- ========================
local function getSlotStroke()
    local ok, stroke = pcall(function()
        local playerGui = LocalPlayer:FindFirstChild("PlayerGui")
        local hotbar    = playerGui
            and playerGui:FindFirstChild("Main")
            and playerGui.Main:FindFirstChild("HUD")
            and playerGui.Main.HUD:FindFirstChild("Hotbar")
        if not hotbar then return nil end

        local slot = nil

        local slotName = hotbarSlotNames[selectedSlot]
        if slotName then slot = hotbar:FindFirstChild(slotName) end

        if not slot then
            for _, child in ipairs(hotbar:GetChildren()) do
                if child:IsA("GuiObject") and child.LayoutOrder == selectedSlot then
                    slot = child
                    break
                end
            end
        end

        if not slot then
            local validSlots = {}
            for _, child in ipairs(hotbar:GetChildren()) do
                if child:IsA("ImageButton") or child:IsA("TextButton") or child.Name:lower():find("slot") then
                    table.insert(validSlots, child)
                end
            end
            table.sort(validSlots, function(a, b)
                return (a.LayoutOrder or 0) < (b.LayoutOrder or 0)
            end)
            slot = validSlots[selectedSlot]
        end

        return slot and slot:FindFirstChildOfClass("UIStroke") or nil
    end)
    return ok and stroke or nil
end

local function isStrokeBlack(stroke)
    if not stroke then return false end
    local c = stroke.Color
    return c.R < 0.1 and c.G < 0.1 and c.B < 0.1
end

-- ========================
-- Auto Equip Logic
-- ========================
local function stopAutoEquip()
    autoEquipEnabled = false
    if strokeConn then strokeConn:Disconnect() strokeConn = nil end
    if autoEquipThread then task.cancel(autoEquipThread) autoEquipThread = nil end
end

local function doEquip()
    if isEquipping then return end
    isEquipping = true
    local kc = slotKeyCode[selectedSlot]
    if kc then
        pressKey(kc)
        task.wait(0.4)
    end
    isEquipping = false
end

local function startAutoEquip()
    stopAutoEquip()
    if not selectedSlot then return end
    autoEquipEnabled = true

    autoEquipThread = task.spawn(function()
        local stroke = nil
        while autoEquipEnabled do
            stroke = getSlotStroke()
            if stroke then break end
            task.wait(0.5)
        end
        if not stroke then return end

        if isStrokeBlack(stroke) then doEquip() end

        strokeConn = stroke:GetPropertyChangedSignal("Color"):Connect(function()
            if not autoEquipEnabled then return end
            if isStrokeBlack(stroke) then doEquip() end
        end)
    end)
end

LocalPlayer.CharacterAdded:Connect(function()
    task.wait(2)
    if autoEquipEnabled then startAutoEquip() end
end)

local function updateFarmState()
    local anyBossEnabled = false
    for _, enabled in pairs(targetBosses) do
        if enabled then anyBossEnabled = true break end
    end

    if not anyBossEnabled then
        farmingEnabled = false
        if farmThread then task.cancel(farmThread) farmThread = nil end
        return
    end

    if farmingEnabled then return end
    farmingEnabled = true

    farmThread = task.spawn(function()
        while farmingEnabled do
            local hasTarget       = false
            local activeBossCount = 0

            for bossName, enabled in pairs(targetBosses) do
                if not enabled then continue end
                activeBossCount = activeBossCount + 1
                local hrp     = getHRP()
                local bossPos = getBossPosition(bossName)
                if hrp and bossPos then
                    hrp.CFrame = CFrame.new(bossPos)
                    hasTarget  = true
                    hasResetThisCycle = false
                    task.wait(0.2)
                end
            end

            if not hasTarget and activeBossCount > 0 then
                if autoResetEnabled and not hasResetThisCycle then
                    local char = LocalPlayer.Character
                    local hum = char and char:FindFirstChildOfClass("Humanoid")
                    if hum and hum.Health > 0 then
                        hasResetThisCycle = true
                        hum.Health = 0
                        task.wait(0.5)
                    end
                end

                if useSavePosEnabled and savedPosition then
                    local hrp = getHRP()
                    if hrp then
                        hrp.CFrame = CFrame.new(savedPosition)
                        task.wait(0.3)
                    end
                end
                task.wait(0.3)
            else
                task.wait(0.1)
            end
        end
    end)
end

local function isBossActive()
    for bossName, enabled in pairs(targetBosses) do
        if enabled and getBossPosition(bossName) then return true end
    end
    return false
end

    autoSkillEnabled = false
    if autoSkillThread then task.cancel(autoSkillThread) autoSkillThread = nil end
end

local function startAutoSkill()
    stopAutoSkill()
    autoSkillEnabled = true
    autoSkillThread = task.spawn(function()
        while autoSkillEnabled do
            if isBossActive() then
                if isEquipping then task.wait(0.1) continue end
                local firedAny = false
                for keyStr, isSelected in pairs(selectedSkills) do
                    if isSelected and Enum.KeyCode[keyStr] then
                        firedAny = true
                        if not isEquipping then
                            task.spawn(function() pressKey(Enum.KeyCode[keyStr]) end)
                        end
                    end
                end
                task.wait(firedAny and 0.3 or 0.1)
            else
                task.wait(0.5)
            end
        end
    end)
end

local function findProximityPrompt(model)
    if not model then return nil end
    for _, v in ipairs(model:GetDescendants()) do
        if v:IsA("ProximityPrompt") then return v end
    end
    return nil

local function scanMobList()
    local newList = {}
    local clientEntities = workspace:FindFirstChild("ClientEntities")
    if clientEntities then
        for _, child in ipairs(clientEntities:GetChildren()) do
            if child:IsA("Model") then
                newList[child.Name] = mobList[child.Name] or false
            end
        end
    end
    mobList = newList
    return mobList
end

local function getMobNames()
    local names = {}
    for name, _ in pairs(mobList) do table.insert(names, name) end
    table.sort(names)
    return names
end

local function clickGuiButton(button)
    pcall(function()
        if getconnections then
            for _, connection in pairs(getconnections(button.MouseButton1Click or button.Activated)) do
                connection:Fire()
            end
        end
        button:SimulateClick()
    end)
end

local function triggerProximityPrompt(prompt)
    if not prompt then return false end
    local ok = pcall(function()
        if fireproximityprompt then
            fireproximityprompt(prompt)
        else
            pressKey(prompt.KeyboardKeyCode)
        end
    end)
    return ok
end

local function parseProgress(progressText)
    if not progressText then return nil, nil end
    local current, max = progressText:match("(%d+)%s*/%s*(%d+)")
    if current and max then return tonumber(current), tonumber(max) end
    return nil, nil
end

local function stopAutoQuest()
    autoQuestEnabled = false
    autoQuestStatus  = "Idle"
    if autoQuestThread then task.cancel(autoQuestThread) autoQuestThread = nil end
end

-- ========================
-- [UPDATED] มหาลูปคำสั่งระบบเควส (1 -> 2 -> 3 -> 4) สมบูรณ์แบบตามพาทที่คุณให้มา
-- ========================
local function startAutoQuest()
    stopAutoQuest()
    if not selectedQuestNpc then
        WindUI:Notify({ Title = "Error", Content = "Please select a Quest NPC first", Duration = 3, Icon = "alert-circle" })
        return
    end
    if not selectedMob then
        WindUI:Notify({ Title = "Error", Content = "Please select a Mob to farm first", Duration = 3, Icon = "alert-circle" })
        return
    end

    autoQuestEnabled = true
    autoQuestThread = task.spawn(function()
        while autoQuestEnabled do
            
            -- [ขั้นตอนที่ 1]: วาปไปข้างหน้าโมเดล NPC ที่เลือกไว้
            autoQuestStatus = "1. Teleporting to NPC Face"
            local hrp = getHRP()
            local npcPos = getQuestNpcPosition(selectedQuestNpc)
            if hrp and npcPos then
                hrp.CFrame = CFrame.new(npcPos + Vector3.new(0, 0, 3))
            end
            task.wait(1.0)
            if not autoQuestEnabled then break end

            -- [ขั้นตอนที่ 2]: ทำการกด E / เปิดการใช้งาน ProximityPrompt
            autoQuestStatus = "2. Pressing E / Triggering Prompt"
            local npcModel  = getQuestNpcModel(selectedQuestNpc)
            local prompt     = npcModel and findProximityPrompt(npcModel)
            if prompt then
                triggerProximityPrompt(prompt)
            end
            task.wait(1.5) -- รอแถบไดอะล็อก Options เปิดแสดงผลครบชิ้น
            if not autoQuestEnabled then break end

            -- [ขั้นตอนที่ 3]: สแกนหาตัวเลือก Options ที่ระบุ (.Main.Label) คลิกช่องเควสที่มี Text ตรงกับชื่อ MOB
            autoQuestStatus = "3. Checking Options for Mob Match"
            local mobSelectedSuccess = false
            local playerGui = LocalPlayer:FindFirstChild("PlayerGui")
            local optionsFolder = playerGui 
                and playerGui:FindFirstChild("Main")
                and playerGui.Main:FindFirstChild("HUD")
                and playerGui.Main.HUD:FindFirstChild("Dialogue")
                and playerGui.Main.HUD.Dialogue:FindFirstChild("Options")

            if optionsFolder then
                for _, child in ipairs(optionsFolder:GetChildren()) do
                    if child:IsA("GuiObject") or child.Name == "Template" then
                        local mainFrame = child:FindFirstChild("Main")
                        local labelObj  = mainFrame and mainFrame:FindFirstChild("Label")
                        
                        if labelObj and labelObj:IsA("TextLabel") and labelObj.Text then
                            if labelObj.Text:lower():find(selectedMob:lower()) then
                                local clickTarget = child:IsA("TextButton") and child or child:FindFirstChildOfClass("TextButton") or child
                                clickGuiButton(clickTarget)
                                mobSelectedSuccess = true
                                print("🎯 [MonHub] Matched & Clicked Quest Slot: " .. labelObj.Text)
                                break
                            end
                        end
                    end
                end
            end
            task.wait(1.0) -- รอจังหวะหน้าจอเปลี่ยนสถานะเป็นหน้าต่างยืนยัน Accept
            if not autoQuestEnabled then break end

            -- [ขั้นตอนกด Accept]: เช็คจนกว่าจะเจอ Properties Text ในโฟลเดอร์เดิมที่เขียนว่า Accept แล้วกดทันที
            autoQuestStatus = "3.5. Scanning for Accept Properties"
            local acceptPressed = false
            for i = 1, 15 do
                if optionsFolder then
                    for _, child in ipairs(optionsFolder:GetChildren()) do
                        local mainFrame = child:FindFirstChild("Main")
                        local labelObj  = mainFrame and mainFrame:FindFirstChild("Label")
                        
                        if labelObj and labelObj:IsA("TextLabel") and labelObj.Text then
                            if labelObj.Text:lower():find("accept") then
                                local clickTarget = child:IsA("TextButton") and child or child:FindFirstChildOfClass("TextButton") or child
                                clickGuiButton(clickTarget)
                                acceptPressed = true
                                print("✅ [MonHub] Accepted properties matched and executed!")
                                break
                            end
                        end
                    end
                end
                if acceptPressed then break end
                task.wait(0.2)
            end
            task.wait(0.8)
            if not autoQuestEnabled then break end

            -- [ขั้นตอนที่ 4]: วาปไปหาโมเดล MOB และตรวจสอบจำนวนค่าความคืบหน้า (QuestBarProgress) ตลอดเวลา
            autoQuestStatus = "4. Farming Mobs & Monitoring Progress"
            local farmTimeout = 0
            
            while autoQuestEnabled do
                local current, max = nil, nil
                
                pcall(function()
                    local questScroll = playerGui.Main.Pages.Quests.Page.Main.Bottom.QuestScroll
                    for _, folder in ipairs(questScroll:GetChildren()) do
                        if folder.Name:find("Repeatable") or folder:FindFirstChild("Main") then
                            local progressObj = folder.Main.Info.QuestBar.QuestBarProgress
                            if progressObj and progressObj:IsA("TextLabel") and progressObj.Text ~= "" then
                                current, max = parseProgress(progressObj.Text)
                                if current and max then break end
                            end
                        end
                    end
                end)

                -- เช็คเงื่อนไขเมื่อเลขเท่ากัน (เช่น 8/8 หรือ 5/5) เพื่อสั่งเคลียร์และวาปกลับไปคุยกับ NPC เพื่อส่งในรอบต่อไป
                if current and max and current >= max then
                    print("🎉 [MonHub] Quest Finished (" .. tostring(current) .. "/" .. tostring(max) .. ") Teleporting back to start next loop!")
                    break
                end

                -- ทำการเคลื่อนย้ายไปตีมอนสเตอร์ตัวที่ระบุ
                local mHrp = getHRP()
                local mobPos = getMobPosition(selectedMob)
                if mHrp and mobPos then
                    mHrp.CFrame = CFrame.new(mobPos)
                    farmTimeout = 0
                else
                    farmTimeout = farmTimeout + 1
                    if farmTimeout > 120 then break end
                end
                task.wait(0.3)
            end
            if not autoQuestEnabled then break end

            -- จบลูปฟาร์ม ย้อนรันข้อ 1 คุยกับ NPC เดิมเพื่อเครดิตรางวัลและเริ่มใหม่
            autoQuestStatus = "Loop Completed! Resetting Task..."
            task.wait(0.2)
        end
        autoQuestStatus = "Idle"
    end)
end

local coreGui = game:GetService("CoreGui")
pcall(function() FloatGui.Parent = coreGui end)
if not FloatGui.Parent or FloatGui.Parent ~= coreGui then
    FloatGui.Parent = LocalPlayer:WaitForChild("PlayerGui")
end

FloatBtn.Name             = "ToggleBtn"
FloatBtn.Parent           = FloatGui
FloatBtn.BackgroundColor3 = Color3.fromRGB(20, 20, 25)

FloatBtn.Size             = UDim2.fromOffset(52, 52)

FloatStroke.Thickness = 2.5
FloatStroke.Parent    = FloatBtn

local dragging  = false
local dragInput = nil
local dragStart = nil
local startPos  = nil

FloatBtn.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1
    or input.UserInputType == Enum.UserInputType.Touch then
        dragging  = true
        dragStart = input.Position
        startPos  = FloatBtn.Position
    end
end)

FloatBtn.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1
    or input.UserInputType == Enum.UserInputType.Touch then
        dragging = false
    end
end)

FloatBtn.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement
    or input.UserInputType == Enum.UserInputType.Touch then
        dragInput = input
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if dragging and dragInput and input == dragInput then
        local delta = input.Position - dragStart
        FloatBtn.Position = UDim2.new(
            startPos.X.Scale, startPos.X.Offset + delta.X,
            startPos.Y.Scale, startPos.Y.Offset + delta.Y
        )
    end
end)

local menuVisible = true

local function toggleMenu()
    menuVisible = not menuVisible
    if menuVisible then
        FloatBtn.Visible = false
        pcall(function()
            if Window.Show then Window:Show()
            elseif Window.Open then Window:Open()
            elseif Window.Toggle then Window:Toggle() end
        end)
    else
        FloatBtn.Visible = true
        pcall(function()
            if Window.Hide then Window:Hide()
            elseif Window.Close then Window:Close()
            elseif Window.Toggle then Window:Toggle() end
        end)
    end
end

FloatBtn.MouseButton1Click:Connect(function()
    if dragStart and startPos then
        local curPos = FloatBtn.Position
        local dx = math.abs(curPos.X.Offset - startPos.X.Offset)
        local dy = math.abs(curPos.Y.Offset - startPos.Y.Offset)
        if dx > 5 or dy > 5 then return end
    end
    menuVisible = false
    toggleMenu()
end)

task.spawn(function()
    task.wait(1)
    while true do
        task.wait(0.4)
        pcall(function()
            local currentUiVisible = false
            local mainFrame = Window.Main or Window.Frame or Window.Container
            if mainFrame then
                currentUiVisible = mainFrame.Visible
            else
                for _, gui in ipairs(coreGui:GetChildren()) do
                    if (gui.Name == "WindUI" or gui.Name == "MonHubSimple") and gui:IsA("ScreenGui") then
                        currentUiVisible = gui.Enabled
                        break
                    end
                end
            end
            
            menuVisible = currentUiVisible
            FloatBtn.Visible = not currentUiVisible
        end)
    end
end)
