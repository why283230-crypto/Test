local function getHRP()

local function pressKey(keyCode)
    pcall(function()
        VirtualInputManager:SendKeyEvent(true,  keyCode, false, game)
        task.wait(0.02)
        VirtualInputManager:SendKeyEvent(false, keyCode, false, game)
    end)
end

local function getBossPosition(bossName)
    local folderName = bossPaths[bossName]
    if not folderName then return nil end
    local clientEntities = workspace:FindFirstChild("ClientEntities")
    if not clientEntities then return nil end
    local bossModel = clientEntities:FindFirstChild(folderName)
    if not bossModel then return nil end
    local root = bossModel:FindFirstChild("HumanoidRootPart")
        or bossModel:FindFirstChild("RootPart")
        or bossModel.PrimaryPart
    if not root then
        for _, part in ipairs(bossModel:GetDescendants()) do
            if part:IsA("BasePart") then root = part break end
        end
    end
    return root and root.Position
end

-- ========================
-- getSlotStroke
-- ========================
local function getSlotStroke()
    local ok, stroke = pcall(function()
        local playerGui = LocalPlayer:FindFirstChild("PlayerGui")
        local hotbar    = playerGui
            and playerGui:FindFirstChild("Main")
            and playerGui.Main:FindFirstChild("HUD")
            and playerGui.Main.HUD:FindFirstChild("Hotbar")
        if not hotbar then return nil end
        local slot = nil
        local slotName = hotbarSlotNames[selectedSlot]
        if slotName then slot = hotbar:FindFirstChild(slotName) end
        if not slot then
            for _, child in ipairs(hotbar:GetChildren()) do
                if child:IsA("GuiObject") and child.LayoutOrder == selectedSlot then
                    slot = child break
                end
            end
        end
        if not slot then
            local validSlots = {}
            for _, child in ipairs(hotbar:GetChildren()) do
                if child:IsA("ImageButton") or child:IsA("TextButton") or child.Name:lower():find("slot") then
                    table.insert(validSlots, child)
                end
            end
            table.sort(validSlots, function(a, b) return (a.LayoutOrder or 0) < (b.LayoutOrder or 0) end)
            slot = validSlots[selectedSlot]
        end
        return slot and slot:FindFirstChildOfClass("UIStroke") or nil
    end)
    return ok and stroke or nil
end

local function isStrokeBlack(stroke)
    if not stroke then return false end
    local c = stroke.Color
    return c.R < 0.1 and c.G < 0.1 and c.B < 0.1
end

-- Fruit Helpers

local function findProximityPromptFruit(model)
    if not model then return nil end
    for _, v in ipairs(model:GetDescendants()) do
        if v:IsA("ProximityPrompt") then return v end
    end
    return nil
end

local function triggerProximityPromptFruit(prompt)
    if not prompt then return false end
    pcall(function()
        if fireproximityprompt then
            fireproximityprompt(prompt)
        else
            local root = prompt.Parent
            if root and root:IsA("BasePart") then
                local hrp = getHRP()
                if hrp then hrp.CFrame = CFrame.new(root.Position + Vector3.new(0, 3, 0)) end
            end
            if holdproximityprompt then
                holdproximityprompt(prompt)
                task.wait(0.5)
                releaseproximityprompt(prompt)
            end
        end
    end)
    return true
end

local function tryGrabFruit(fruitName)
    local model = workspace:FindFirstChild(fruitName)
    if not model then return false end
    local hrp = getHRP()
    if not hrp then return false end

    local targetPart = model.PrimaryPart
        or model:FindFirstChild("HumanoidRootPart")
        or model:FindFirstChild("RootPart")
    if not targetPart then
        for _, part in ipairs(model:GetDescendants()) do
            if part:IsA("BasePart") then targetPart = part break end
        end
    end
    if not targetPart then return false end

    hrp.CFrame = CFrame.new(targetPart.Position + Vector3.new(0, 3, 0))
    WindUI:Notify({ Title = "Found: " .. fruitName, Content = "Teleported! Waiting 3s before grabbing...", Duration = 3, Icon = "zap" })
    task.wait(3)

    local stillExists = workspace:FindFirstChild(fruitName)
    if not stillExists then
        WindUI:Notify({ Title = fruitName .. " disappeared!", Duration = 2, Icon = "x" })
        return false
    end

    local prompt = findProximityPromptFruit(stillExists)
    if prompt then
        triggerProximityPromptFruit(prompt)
        WindUI:Notify({ Title = "Grabbed: " .. fruitName, Content = "ProximityPrompt triggered!", Duration = 3, Icon = "check-circle" })
        return true
    else
        WindUI:Notify({ Title = "No Prompt on " .. fruitName, Duration = 2, Icon = "alert-circle" })
        return false
    end
end

local function grabAllPendingFruits()
    for _, fruit in ipairs(fruitTargets) do
        if fruit.enabled then
            local model = workspace:FindFirstChild(fruit.name)
            if model then
                tryGrabFruit(fruit.name)
                task.wait(1)
            end
        end
    end
end

-- ========================
-- Auto Grab Loop
-- ========================
local function stopAutoGrab()
    autoGrabEnabled = false
    if autoGrabThread then task.cancel(autoGrabThread) autoGrabThread = nil end
end

local function startAutoGrab()
    stopAutoGrab()
    autoGrabEnabled = true
    autoGrabThread = task.spawn(function()
        while autoGrabEnabled do
            local found = false
            if farmingEnabled then
                grabAllPendingFruits()
                task.wait(2)
            else
                for _, fruit in ipairs(fruitTargets) do
                    if not autoGrabEnabled then break end
                    if fruit.enabled then
                        local model = workspace:FindFirstChild(fruit.name)
                        if model then
                            found = true
                            tryGrabFruit(fruit.name)
                            task.wait(2)
                        end
                    end
                end
                if not found then task.wait(1) end
            end
        end
    end)
end

-- ========================
-- Auto Equip Logic
-- ========================
local function stopAutoEquip()
    autoEquipEnabled = false
    if strokeConn then strokeConn:Disconnect() strokeConn = nil end
    if autoEquipThread then task.cancel(autoEquipThread) autoEquipThread = nil end
end

local function doEquip()
    if isEquipping then return end
    isEquipping = true
    local kc = slotKeyCode[selectedSlot]
    if kc then pressKey(kc) task.wait(0.4) end
    isEquipping = false
end

local function startAutoEquip()
    stopAutoEquip()
    if not selectedSlot then return end
    autoEquipEnabled = true
    autoEquipThread = task.spawn(function()
        local stroke = nil
        while autoEquipEnabled do
            stroke = getSlotStroke()
            if stroke then break end
            task.wait(0.5)
        end
        if not stroke then return end
        if isStrokeBlack(stroke) then doEquip() end
        strokeConn = stroke:GetPropertyChangedSignal("Color"):Connect(function()
            if not autoEquipEnabled then return end
            if isStrokeBlack(stroke) then doEquip() end
        end)
    end)
end

LocalPlayer.CharacterAdded:Connect(function()
    task.wait(2)
    if autoEquipEnabled then startAutoEquip() end
end)

-- ========================
-- Farm Logic (Main Boss Farm)
-- หมายเหตุ: ระบบนี้จะ "หยุดให้สิทธิ์" การควบคุม HRP ทันทีที่ Train x4 เปิดใช้
-- เพื่อไม่ให้สองระบบแย่งกันเซต CFrame จนบัค (ดู guard ด้านล่าง)
-- ========================
local function updateFarmState()
    local anyBossEnabled = false
    for _, enabled in pairs(targetBosses) do
        if enabled then anyBossEnabled = true break end
    end

    if not anyBossEnabled then
        farmingEnabled = false
        if farmThread then task.cancel(farmThread) farmThread = nil end
        return
    end

    if farmingEnabled then return end
    farmingEnabled = true

    farmThread = task.spawn(function()
        while farmingEnabled do
            -- Guard: ถ้า Train x4 เปิดอยู่ ให้ระบบฟาร์มบอสหยุดขยับ HRP ชั่วคราว
            if trainX4Enabled then
                task.wait(0.5)
                continue
            end

            grabAllPendingFruits()

            local hasTarget       = false
            local activeBossCount = 0

            for bossName, enabled in pairs(targetBosses) do
                if not enabled then continue end
                activeBossCount = activeBossCount + 1
                local hrp     = getHRP()
                local bossPos = getBossPosition(bossName)
                if hrp and bossPos then
                    hrp.CFrame = CFrame.new(bossPos)
                    hasTarget  = true
                    hasResetThisCycle = false
                    task.wait(0.2)
                end
            end

            if not hasTarget and activeBossCount > 0 then
                if autoResetEnabled and not hasResetThisCycle then
                    local char = LocalPlayer.Character
                    local hum = char and char:FindFirstChildOfClass("Humanoid")
                    if hum and hum.Health > 0 then
                        hasResetThisCycle = true
                        hum.Health = 0
                        task.wait(0.5)
                    end
                end
                if useSavePosEnabled and savedPosition then
                    local hrp = getHRP()
                    if hrp then hrp.CFrame = CFrame.new(savedPosition) task.wait(0.3) end
                end
                task.wait(0.3)
            else
                task.wait(0.1)
            end
        end
    end)
end

local function isBossActive()
    for bossName, enabled in pairs(targetBosses) do
        if enabled and getBossPosition(bossName) then return true end
    end
    return false
end

-- ========================
-- Auto Skill Logic
-- หมายเหตุ: หยุดกดสกิลตอน Train x4 ทำงานเพื่อไม่ให้ปนกัน
-- ========================
local function stopAutoSkill()
    autoSkillEnabled = false
    if autoSkillThread then task.cancel(autoSkillThread) autoSkillThread = nil end
end

local function startAutoSkill()
    stopAutoSkill()
    autoSkillEnabled = true
    autoSkillThread = task.spawn(function()
        while autoSkillEnabled do
            if trainX4Enabled then
                task.wait(0.3)
                continue
            end
            if isBossActive() then
                if isEquipping then task.wait(0.1) continue end
                local firedAny = false
                for keyStr, isSelected in pairs(selectedSkills) do
                    if isSelected and Enum.KeyCode[keyStr] then
                        firedAny = true
                        if not isEquipping then
                            task.spawn(function() pressKey(Enum.KeyCode[keyStr]) end)
                        end
                    end
                end
                task.wait(firedAny and 0.3 or 0.1)
            else
                task.wait(0.5)
            end
        end
    end)
end

-- ========================
-- Server Hop Logic
-- ========================
local function serverHop()
    WindUI:Notify({ Title = "Server Hop", Content = "Finding a new server...", Duration = 3, Icon = "refresh-cw" })
    task.wait(1)
    pcall(function()
        local servers = HttpService:JSONDecode(game:HttpGet(
            "https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=Asc&limit=100"
        ))
        for _, server in ipairs(servers.data) do
            if server.playing < server.maxPlayers and server.id ~= game.JobId then
                TeleportService:TeleportToPlaceInstance(game.PlaceId, server.id, LocalPlayer)
                break
            end
        end
    end)
end

local function stopAutoHop()
    autoHopEnabled = false
    if autoHopThread then task.cancel(autoHopThread) autoHopThread = nil end
end

local function startAutoHop()
    stopAutoHop()
    autoHopEnabled = true
    hopCountdown = hopInterval
    autoHopThread = task.spawn(function()
        while autoHopEnabled do
            task.wait(60)
            if not autoHopEnabled then break end
            hopCountdown = hopCountdown - 1
            if hopCountdown <= 0 then
                hopCountdown = hopInterval
                serverHop()
            end
        end
    end)
end

-- ========================
-- Teleport Zone Helpers
-- workspace.Zones.Training -> สแกนชื่อโซนทั้งหมด
-- ========================
local function scanTrainingZones()
    local zones = {}
    local ok, trainingFolder = pcall(function()
        return workspace.Zones.Training
    end)
    if ok and trainingFolder then
        for _, child in ipairs(trainingFolder:GetChildren()) do
            table.insert(zones, child.Name)
        end
    end
    if #zones == 0 then table.insert(zones, "No Zones Found") end
    table.sort(zones)
    availableZones = zones
    return zones
end

local function getZonePosition(zoneName)
    local ok, trainingFolder = pcall(function()
        return workspace.Zones.Training
    end)
    if not ok or not trainingFolder then return nil end
    local zoneObj = trainingFolder:FindFirstChild(zoneName)
    if not zoneObj then return nil end

    if zoneObj:IsA("BasePart") then
        return zoneObj.Position
    end

    -- ถ้าเป็น Model / Folder ให้หา BasePart ข้างใน
    local root = zoneObj:FindFirstChild("HumanoidRootPart")
        or zoneObj:FindFirstChild("RootPart")
        or (zoneObj.PrimaryPart)
    if not root then
        for _, part in ipairs(zoneObj:GetDescendants()) do
            if part:IsA("BasePart") then root = part break end
        end
    end
    return root and root.Position or nil
end

-- ========================
-- Train x4 Clan Point Logic
-- ล็อค Position ของ Player ตอนเปิดใช้ + กด Slot 4 + เดิน/กระโดดอัตโนมัติ
-- หมายเหตุ: ระบบนี้ "exclusive" กับ Farm บอส/Auto Skill เพื่อไม่ให้แย่ง HRP กัน (ดู guard ด้านบน)
-- ========================
local function stopTrainX4()
    trainX4Enabled = false
    lockedTrainPos = nil
    if trainX4Thread then task.cancel(trainX4Thread) trainX4Thread = nil end
end

local function startTrainX4()
    stopTrainX4()
    local hrp = getHRP()
    if not hrp then
        WindUI:Notify({ Title = "Error", Content = "Character not found", Duration = 3, Icon = "alert-circle" })
        return
    end

    -- ล็อค Position ปัจจุบันไว้ตอนเปิดใช้
    lockedTrainPos = hrp.CFrame
    trainX4Enabled = true

    -- กด Slot 4 ทันทีตอนเริ่ม
    pressKey(Enum.KeyCode.Four)

    trainX4Thread = task.spawn(function()
        local char = LocalPlayer.Character
        local hum  = char and char:FindFirstChildOfClass("Humanoid")

        while trainX4Enabled do
            local curHRP = getHRP()
            if curHRP and lockedTrainPos then
                -- ล็อคตำแหน่งไว้ที่เดิมเสมอ (กันไหลจากแรงกระแทก/knockback)
                curHRP.CFrame = lockedTrainPos
            end

            -- จำลองกดเดิน (W) และกระโดด (Space) อัตโนมัติ
            pressKey(Enum.KeyCode.W)
            task.wait(0.1)
            if hum then
                hum.Jump = true
            else
                pressKey(Enum.KeyCode.Space)
            end

            task.wait(0.4)
        end
    end)
end

local dragging  = false
local dragInput = nil
local dragStart = nil
local startPos  = nil

FloatBtn.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1
    or input.UserInputType == Enum.UserInputType.Touch then
        dragging  = true
        dragStart = input.Position
        startPos  = FloatBtn.Position
    end
end)

FloatBtn.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1
    or input.UserInputType == Enum.UserInputType.Touch then
        dragging = false
    end
end)

FloatBtn.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement
    or input.UserInputType == Enum.UserInputType.Touch then
        dragInput = input
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if dragging and dragInput and input == dragInput then
        local delta = input.Position - dragStart
        FloatBtn.Position = UDim2.new(
            startPos.X.Scale, startPos.X.Offset + delta.X,
            startPos.Y.Scale, startPos.Y.Offset + delta.Y
        )
    end
end)

local menuVisible = true

local function toggleMenu()
    menuVisible = not menuVisible
    if menuVisible then
        FloatBtn.Visible = false
        pcall(function()
            if Window.Show then Window:Show()
            elseif Window.Open then Window:Open()
            elseif Window.Toggle then Window:Toggle() end
        end)
    else
        FloatBtn.Visible = true
        pcall(function()
            if Window.Hide then Window:Hide()
            elseif Window.Close then Window:Close()
            elseif Window.Toggle then Window:Toggle() end
        end)
    end
end

FloatBtn.MouseButton1Click:Connect(function()
    if dragStart and startPos then
        local curPos = FloatBtn.Position
        local dx = math.abs(curPos.X.Offset - startPos.X.Offset)
        local dy = math.abs(curPos.Y.Offset - startPos.Y.Offset)
        if dx > 5 or dy > 5 then return end
    end
    menuVisible = false
    toggleMenu()
end)

task.spawn(function()
    task.wait(1)
    while true do
        task.wait(0.4)
        pcall(function()
            local currentUiVisible = false
            local mainFrame = Window.Main or Window.Frame or Window.Container
            if mainFrame then
                currentUiVisible = mainFrame.Visible
            else
                for _, gui in ipairs(coreGui:GetChildren()) do
                    if (gui.Name == "WindUI" or gui.Name == "MonHubSimple") and gui:IsA("ScreenGui") then
                        currentUiVisible = gui.Enabled
                        break
                    end
                end
            end
            menuVisible = currentUiVisible
            FloatBtn.Visible = not currentUiVisible
        end)
    end
end)
